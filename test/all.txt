# cards.js (dans static/js/)
// Fonction pour créer les cartes
const createCard = (item) => {
    const card = document.createElement("div");
    card.classList.add("card");
    card.innerHTML = `
        <img src="${item.image}" alt="${item.location}" class="card-image">
        <!-- reste du HTML de la carte -->
    `;
    return card;
};

// Fonction pour initialiser les cartes
function initializeCards(data) {
    const container = document.querySelector(".container");
    data.forEach(item => {
        const card = createCard(item);
        container.appendChild(card);
    });
}

from django.shortcuts import render
import json

def vue_voyages(request):
    donnees_voyage = [
        {
            "image": "imageVamos/nice.jpg",
            "city": "Nice",

            # ... reste des données
        }
    ]
    return render(request, 'votre_template.html', {

    })

# votre_template.html
{% load static %}
<!DOCTYPE html>
<html>
<head>
    <title>Voyages</title>
</head>
<body>
    <div class="container"></div>

    <!-- Chargement du fichier JS -->
    <script src="{% static 'js/cards.js' %}"></script>
    
    <!-- Script pour passer les données de Django vers JS -->
    <script>
        document.addEventListener("DOMContentLoaded", () => {
            const data = JSON.parse('{{ donnees_voyage|safe }}');
            initializeCards(data);
        });
    </script>
</body>
</html>
